中序遍历
```c++
vector<int> inorderTraversal(TreeNode* root) {
        
        vector<int> answer;        
        stack<TreeNode*> mstack;
        
        while(1)
        {
            while(root)
            {
                mstack.push(root);
                root=root->left;
            }
            
            if(!mstack.size())
                break;
            
            root=mstack.top();
            mstack.pop();
            
            answer.push_back(root->val);
            root=root->right;
        }
        
        
        return answer;
        
    }
```

    vector<int> postorderTraversal(TreeNode* root) {
        
        stack<TreeNode*> mstack;
        vector<int> ans;
        
        if(!root)
            return ans;
        
        while(1)
        {
            while(root)
            {
                if(root->left)
                    mstack.push(root->left);
                ans.push_back(root->val);             
                root=root->right;
            }
            
            if(mstack.empty())
                break;
            else
            {
                root=mstack.top();
                mstack.pop();
            }
                
        }
           
        int i=0;
        int j=ans.size()-1;
        while(i<j)
            swap(ans[i++],ans[j--]);
        
        return ans;
    }
